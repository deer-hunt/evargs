Example code¶

from evargs import EvArgs

evargs = EvArgs()

evargs.initialize({
   'a': {'type': int},
   'b': {'type': str},
   'c': {'type': int, 'list': True},
})

evargs.parse('a=1;b=abc;c=1,2,3;')

assert evargs.get('a') == 1
assert evargs.get('b') == 'abc'
assert evargs.get('c') == [1, 2, 3]
